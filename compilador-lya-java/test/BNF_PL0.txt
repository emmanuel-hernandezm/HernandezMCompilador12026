<Programa> ::= <Bloque> "."

<Bloque> ::= <Seccion_Const> <Seccion_Var> <Seccion_Proced> <Proposición>

<Seccion_Const> ::= "const" <Lista_Const> ";" | ε
<Lista_Const>   ::= id "=" num | <Lista_Const> "," id "=" num

<Seccion_Var> ::= "var" <Lista_Var> ";" | ε
<Lista_Var>   ::= id | <Lista_Var> "," id

<Seccion_Proced> ::= <Lista_Proced> | ε
<Lista_Proced>   ::= <Decl_Proced> | <Lista_Proced> <Decl_Proced>
<Decl_Proced>    ::= "proced" id ";" <Bloque> ";"


<Condición> ::= <Expresión> <Operador_Relacional> <Expresión>

<Operador_Relacional> ::= "==" | "!=" | "<" | ">" | "<=" | ">="

<Proposición> ::= "begin" <Lista_Proposiciones> "end"
                | id "=" <Expresión>
                | "write" <Argumento_Write>
                | "read" id
                | "call" id
                | "if" <Condición> "then" <Proposición>
                | "while" <Condición> "do" <Proposición>
                | "for" id "=" <Expresión> <Direccion_For> <Expresión> "do" <Proposición>

<Lista_Proposiciones> ::= <Proposición> | <Lista_Proposiciones> ";" <Proposición>

<Argumento_Write> ::= id | num

<Direccion_For> ::= "to" | "down"

<Expresión> ::= <Término> 
              | <Expresión> "+" <Término> 
              | <Expresión> "-" <Término>

<Término> ::= <Factor> 
            | <Término> "*" <Factor> 
            | <Término> "/" <Factor>

<Factor> ::= "(" <Expresión> ")" 
           | id 
           | num
